def virgule(listeTransposer : List[List[str]], listeDonnee : List[List[str]])-> List[List[str]]:
    """!
    fonction qui remplace les virgules par des points
    @param listeTransposer List la liste des valeurs
    @param listeDonnee liste permettant d'indiquer la taille
    @return List la liste des listes de valeurs sans les virgules
    """
    for i in range(1,8):
        for j in range (0,len(listeDonnee)):
            chiffre = listeTransposer[i][j]
            new = chiffre.replace(',', '.')
            listeTransposer[i][j] = new
    return listeTransposer
